🔧 Panduan Setup Google Sheets

Langkah 1. Buka spreadsheet → klik Extensions → Apps Script.

Langkah 2. Hapus kode lama, paste kode ini (sudah mendukung penyimpanan konfigurasi):

const SHEET_ID = "1ncWuZxFDaDqNFYNn2Co052JvGYYBdSn-nzrcBhi4jZ8"; const SHEET_NAME = "Sheet1"; const CFG_SHEET = "Konfigurasi"; // sheet baru untuk config function getOrCreateCfgSheet(ss) { let sh = ss.getSheetByName(CFG_SHEET); if (!sh) { sh = ss.insertSheet(CFG_SHEET); sh.getRange(1,1,1,2).setValues([["key","value"]]); } return sh; } function doPost(e) { try { const ss = SpreadsheetApp.openById(SHEET_ID); const body = JSON.parse(e.postData.contents); // — simpan konfigurasi — if (body.__type === "config") { const sh = getOrCreateCfgSheet(ss); const cfg = body.data; const keys = Object.keys(cfg); const all = sh.getDataRange().getValues(); keys.forEach(k => { let found = false; for (let i = 1; i < all.length; i++) { if (all[i][0] === k) { sh.getRange(i+1,2).setValue(cfg[k]); found = true; break; } } if (!found) sh.appendRow([k, cfg[k]]); }); return json({status:"ok", saved: keys.length}); } // — simpan nilai siswa — const sh = ss.getSheetByName(SHEET_NAME); const rows = body; const last = sh.getLastRow(); if (last > 1) sh.getRange(2,1,last-1,15).clearContent(); rows.forEach((r,i) => { sh.getRange(i+2,1,1,15).setValues([[ r.no,r.nama,r.nis,r.nisn,r.jk, r.f1,r.f2,r.f3,r.avgF, r.s1,r.s2,r.s3,r.avgS, r.sas,r.rapor ]]); }); return json({status:"ok", rows:rows.length}); } catch(err) { return json({status:"error", msg:err.toString()}); } } function doGet(e) { const ss = SpreadsheetApp.openById(SHEET_ID); // — baca konfigurasi — if (e.parameter.action === "getConfig") { const sh = getOrCreateCfgSheet(ss); const all = sh.getDataRange().getValues().slice(1); const cfg = {}; all.forEach(r => { if (r[0]) cfg[r[0]] = r[1]; }); return json(cfg); } // — simpan via GET fallback — if (e.parameter.data) { try { const sh = ss.getSheetByName(SHEET_NAME); const rows = JSON.parse(decodeURIComponent(e.parameter.data)); const last = sh.getLastRow(); if (last > 1) sh.getRange(2,1,last-1,15).clearContent(); rows.forEach((r,i) => { sh.getRange(i+2,1,1,15).setValues([[ r.no,r.nama,r.nis,r.nisn,r.jk, r.f1,r.f2,r.f3,r.avgF, r.s1,r.s2,r.s3,r.avgS, r.sas,r.rapor ]]); }); return json({status:"ok", rows:rows.length}); } catch(err) { return json({status:"error", msg:err.toString()}); } } // — baca nilai siswa — const sh = ss.getSheetByName(SHEET_NAME); const data = sh.getDataRange().getValues().slice(1) .filter(r=>r[1]).map(r=>({ no:r[0],nama:r[1],nis:r[2],nisn:r[3],jk:r[4], f1:r[5],f2:r[6],f3:r[7],avgF:r[8], s1:r[9],s2:r[10],s3:r[11],avgS:r[12], sas:r[13],rapor:r[14] })); return json(data); } function json(obj) { return ContentService .createTextOutput(JSON.stringify(obj)) .setMimeType(ContentService.MimeType.JSON); }

Langkah 3. Deploy → New Deployment → Web App. Set Execute as: Me & Who has access: Anyone. Copy URL.

Langkah 4. Cari baris const GAS_URL = ""; dan paste URL Anda.

💡 Konfigurasi (nama sekolah, mapel, kelas, dll) kini disimpan di sheet Konfigurasi dan otomatis dimuat di perangkat mana pun.
